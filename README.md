## Example

```javascript
import React, { Component } from "react";

import { IphoneX } from "../../src";

class App extends Component {
  render() {
    return (
      <div>
        <IphoneX
          background={["https://bit.ly/2T8fDWo"]}
          sizes={[0.9, 0.8, 0.5]}
          delay={10000}
          landscape
        />
      </div>
    );
  }
}

export default App;
```
